47 std::unordered_set<std::size_t> PartialBenchmarkDescription::getCipherParamPositions(std::uint32_t cipher_param_mask)
48 {
49  std::unordered_set<std::size_t> retval;
50  std::bitset<sizeof(decltype(cipher_param_mask)) * 8> cipher_param_bits(cipher_param_mask);
51  for (std::size_t i = 0; i < cipher_param_bits.size(); ++i)
52  if (cipher_param_bits.test(i))
53  retval.insert(i);
54  return retval;

125 IBenchmarkDescription::DescriptionToken::Ptr PartialBenchmarkDescription::matchBenchmarkDescriptor(const Engine &engine,
126  const IBenchmarkDescription::BenchmarkConfig &bench_config,
127  const hebench::APIBridge::Handle &h_desc,
128  const std::vector<hebench::APIBridge::WorkloadParam> &w_params) const
129 {
130  IBenchmarkDescription::DescriptionToken::Ptr retval;
131 
132  hebench::APIBridge::BenchmarkDescriptor bench_desc;
133  std::uint64_t w_params_count, other;
134  engine.validateRetCode(hebench::APIBridge::getWorkloadParamsDetails(engine.handle(), h_desc, &w_params_count, &other));
135  if (w_params_count != w_params.size())
136  {
137  std::stringstream ss;
138  ss << "Invalid number of workload arguments. Expected " << w_params_count << ", but " << w_params.size() << " received.";
139  throw std::runtime_error(IL_LOG_MSG_CLASS(ss.str()));
140  } // end if
141  engine.validateRetCode(hebench::APIBridge::describeBenchmark(engine.handle(), h_desc, &bench_desc, nullptr));
142 
143  std::string s_workload_name = matchBenchmarkDescriptor(bench_desc, w_params);
144  if (!s_workload_name.empty())
145  {
146  retval = createToken(bench_config, h_desc, bench_desc, w_params);
147  retval->description.workload_name = s_workload_name;
148  describe(engine, retval);
149  } // end if
150 
151  return retval;
152 }

154 void PartialBenchmarkDescription::describe(const Engine &engine,
155  DescriptionToken::Ptr pre_token) const
156 {
157  Description &description = pre_token->description;
158  hebench::APIBridge::Handle h_bench_desc = pre_token->getDescriptorHandle(this);
159  const hebench::APIBridge::BenchmarkDescriptor &bench_desc = pre_token->getDescriptor(this);
160  const std::vector<hebench::APIBridge::WorkloadParam> &w_params = pre_token->getWorkloadParams(this);
161 
162  std::stringstream ss;
163  std::filesystem::path ss_path;
164  std::string &s_workload_name = description.workload_name;
165  std::string s_path_workload_name;
166  std::string s_scheme_name = engine.getSchemeName(bench_desc.scheme);
167  std::string s_security_name = engine.getSecurityName(bench_desc.scheme, bench_desc.security);
168 
169  // generate path
170  ss = std::stringstream();
171  if (!s_workload_name.empty())
172  ss << s_workload_name << "_";
173  else
174  s_workload_name = std::to_string(static_cast<int>(bench_desc.workload));
175  ss << std::to_string(static_cast<int>(bench_desc.workload));
176  s_path_workload_name = hebench::Utilities::convertToDirectoryName(ss.str());
177  ss_path = s_path_workload_name;
178  ss = std::stringstream();
179  ss << "wp";
180  for (std::size_t i = 0; i < w_params.size(); ++i)
181  {
182  ss << "_";
183  switch (w_params[i].data_type)
184  {
185  case hebench::APIBridge::WorkloadParamType::UInt64:
186  ss << w_params[i].u_param;
187  break;
188 
189  case hebench::APIBridge::WorkloadParamType::Float64:
190  ss << w_params[i].f_param;
191  break;
192 
193  default:
194  ss << w_params[i].i_param;
195  break;
196  } // end switch
197  } // end for
198  ss_path /= hebench::Utilities::convertToDirectoryName(ss.str()); // workload params
199  ss_path /= hebench::Utilities::convertToDirectoryName(PartialBenchmarkDescription::getCategoryName(bench_desc.category));
200  ss_path /= hebench::Utilities::convertToDirectoryName(PartialBenchmarkDescription::getDataTypeName(bench_desc.data_type));
201  std::size_t max_non_zero = HEBENCH_MAX_CATEGORY_PARAMS;
202  while (max_non_zero > 0 && bench_desc.cat_params.reserved[max_non_zero - 1] == 0)
203  --max_non_zero;
204  ss = std::stringstream();
205  if (max_non_zero > 0)
206  {
207  for (std::size_t i = 0; i < max_non_zero; ++i)
208  ss << bench_desc.cat_params.reserved[i];
209  } // end if
210  else
211  ss << "default";
212  ss_path /= ss.str();
213  // cipher/plain parameters
214  ss = std::stringstream();
215  std::unordered_set<std::size_t> cipher_param_pos =
216  PartialBenchmarkDescription::getCipherParamPositions(bench_desc.cipher_param_mask);
217  if (cipher_param_pos.empty())
218  ss << "all_plain";
219  else if (cipher_param_pos.size() >= sizeof(std::uint32_t) * 8)
220  ss << "all_cipher";
221  else
222  {
223  std::size_t max_elem = *std::max_element(cipher_param_pos.begin(), cipher_param_pos.end());
224  for (std::size_t i = 0; i <= max_elem; ++i)
225  ss << (cipher_param_pos.count(i) > 0 ? 'c' : 'p');
226  } // end else
227  ss_path /= ss.str();
228  ss_path /= hebench::Utilities::convertToDirectoryName(s_scheme_name);
229  ss_path /= hebench::Utilities::convertToDirectoryName(s_security_name);
230  ss_path /= std::to_string(bench_desc.other);
231 
232  // generate header
233  ss = std::stringstream();
234  ss << "Specifications," << std::endl
235  << ", Encryption, " << std::endl
236  << ", , Scheme, " << s_scheme_name << std::endl
237  << ", , Security, " << s_security_name << std::endl
238  << ", Extra, " << bench_desc.other << std::endl;
239  std::string s_tmp = engine.getExtraDescription(h_bench_desc, w_params);
240  if (!s_tmp.empty())
241  ss << s_tmp;
242  ss << std::endl
243  << std::endl
244  << ", Category, " << PartialBenchmarkDescription::getCategoryName(bench_desc.category) << std::endl;
245  switch (bench_desc.category)
246  {
247  case hebench::APIBridge::Category::Latency:
248  ss << ", , Warmup iterations, " << bench_desc.cat_params.latency.warmup_iterations_count << std::endl
249  << ", , Minimum test time requested (ms), " << bench_desc.cat_params.latency.min_test_time_ms << std::endl;
250  break;
251 
252  case hebench::APIBridge::Category::Offline:
253  {
254  ss << ", , Parameter, Samples requested" << std::endl;
255 
256  bool all_params_zero = true;
257  for (std::size_t i = 0; i < HEBENCH_MAX_OP_PARAMS; ++i)
258  {
259  if (bench_desc.cat_params.offline.data_count[i] != 0)
260  {
261  all_params_zero = false;
262  ss << ", , " << i << ", " << bench_desc.cat_params.offline.data_count[i] << std::endl;
263  } // end if
264  } // end if
265  if (all_params_zero)
266  ss << ", , All, 0" << std::endl;
267  }
268  break;
269 
270  default:
271  throw std::invalid_argument(IL_LOG_MSG_CLASS("Unsupported benchmark category: " + std::to_string(bench_desc.category) + "."));
272  break;
273  } // end switch
274 
275  ss << std::endl
276  << ", Workload, " << s_workload_name << std::endl
277  << ", , Data type, " << PartialBenchmarkDescription::getDataTypeName(bench_desc.data_type) << std::endl
278  << ", , Encrypted op parameters (index)";
279  if (cipher_param_pos.empty())
280  ss << ", None" << std::endl;
281  else if (cipher_param_pos.size() >= sizeof(std::uint32_t) * 8)
282  ss << ", All" << std::endl;
283  else
284  {
285  // output the indices of the encrypted parameters
286  std::vector<std::size_t> cipher_params(cipher_param_pos.begin(), cipher_param_pos.end());
287  std::sort(cipher_params.begin(), cipher_params.end());
288  for (auto param_index : cipher_params)
289  ss << ", " << param_index;
290  ss << std::endl;
291  } // end else
292 
293  description.header = ss.str();
294  description.path = ss_path;
295 
296  completeDescription(engine, pre_token);
297 }

346 void PartialBenchmark::initBackend(hebench::Utilities::TimingReportEx &out_report, const FriendPrivateKey &)
347 {
348  hebench::Common::EventTimer timer;
349  hebench::Common::TimingReportEvent::Ptr p_timing_event;
350 
351  hebench::APIBridge::WorkloadParams params;
352  params.count = m_workload_params.size();
353  params.params = m_workload_params.data();
354 
355  std::cout << IOS_MSG_INFO << hebench::Logging::GlobalLogger::log("Initializing backend benchmark...") << std::endl;
356  timer.start();
357  validateRetCode(hebench::APIBridge::initBenchmark(m_p_engine->handle(),
358  m_h_descriptor,
359  m_workload_params.empty() ? nullptr : &params,
360  &m_handle));
361  p_timing_event = timer.stop<DefaultTimeInterval>(getEventIDNext(), 1, nullptr);
362  out_report.addEvent<DefaultTimeInterval>(p_timing_event, std::string("Initialization"));
363  hebench::Logging::GlobalLogger::log("OK");
364  std::cout << IOS_MSG_OK << std::endl;
365  std::stringstream ss = std::stringstream();
366  ss << "Elapsed wall time: " << p_timing_event->elapsedWallTime<std::milli>() << " ms";
367  std::cout << IOS_MSG_INFO << hebench::Logging::GlobalLogger::log(ss.str()) << std::endl;
368  ss = std::stringstream();
369  ss << "Elapsed CPU time: " << p_timing_event->elapsedCPUTime<std::milli>() << " ms";
370  std::cout << IOS_MSG_INFO << hebench::Logging::GlobalLogger::log(ss.str()) << std::endl;
371 }
