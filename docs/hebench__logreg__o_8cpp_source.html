49 void BenchmarkDescription::completeDescription(const Engine &engine,
50  DescriptionToken::Ptr pre_token) const
51 {
52  (void)engine;
53 
54  // finish describing workload
55  assert(OpParameterCount == 3);
56  assert(DefaultBatchSize == 100);
57 
58  std::uint64_t vector_size;
59  std::uint64_t default_batch_size = pre_token->getBenchmarkConfiguration(this).default_sample_size > 0 ?
60  pre_token->getBenchmarkConfiguration(this).default_sample_size :
61  BenchmarkDescription::DefaultBatchSize;
62  std::uint64_t batch_sizes[OpParameterCount];
63  std::stringstream ss;
64  hebench::APIBridge::BenchmarkDescriptor bench_desc = pre_token->getDescriptor(this);
65 
66  if (bench_desc.cat_params.offline.data_count[DataGenerator::Index_W] == 0)
67  bench_desc.cat_params.offline.data_count[DataGenerator::Index_W] = 1;
68  if (bench_desc.cat_params.offline.data_count[DataGenerator::Index_b] == 0)
69  bench_desc.cat_params.offline.data_count[DataGenerator::Index_b] = 1;
70 
71  vector_size = fetchVectorSize(pre_token->getWorkloadParams(this));
72 
73  std::uint64_t result_batch_size = computeSampleSizes(batch_sizes,
74  OpParameterCount,
75  default_batch_size,
76  bench_desc);
77  if (batch_sizes[DataGenerator::Index_W] != 1)
78  {
79  ss = std::stringstream();
80  ss << "Batch size for feature vector 'W' must be 1, but " << batch_sizes[DataGenerator::Index_W] << " received.";
81  throw std::invalid_argument(ss.str());
82  } // end if
83  if (batch_sizes[DataGenerator::Index_b] != 1)
84  {
85  ss = std::stringstream();
86  ss << "Batch size for bias 'b' must be 1, but " << batch_sizes[DataGenerator::Index_b] << " received.";
87  throw std::invalid_argument(ss.str());
88  } // end if
89 
90  ss = std::stringstream();
91  ss << pre_token->description.header;
92 
93  // complete header with workload specifics
94  ss << ", , P(X = X') = sigmoid";
95  switch (bench_desc.workload)
96  {
97  case hebench::APIBridge::Workload::LogisticRegression_PolyD3:
98  ss << "_pd3";
99  break;
100  case hebench::APIBridge::Workload::LogisticRegression_PolyD5:
101  ss << "_pd5";
102  break;
103  case hebench::APIBridge::Workload::LogisticRegression_PolyD7:
104  ss << "_pd7";
105  break;
106  default:
107  // standard sigmoid
108  break;
109  } // end switch
110  ss << "(W . X + b)" << std::endl
111  << ", , , Elements, Batch size" << std::endl;
112  ss << ", , W, " << vector_size << ", " << batch_sizes[DataGenerator::Index_W] << std::endl;
113  ss << ", , b, 1, " << batch_sizes[DataGenerator::Index_b] << std::endl;
114  ss << ", , X, " << vector_size << ", " << batch_sizes[DataGenerator::Index_X] << std::endl;
115  ss << ", , P(X), 1, " << result_batch_size << std::endl;
116 
117  pre_token->description.header = ss.str();
118 }

159 void Benchmark::init(const IBenchmarkDescription::Description &description)
160 {
161  (void)description;
162  hebench::Common::EventTimer timer;
163  hebench::Common::TimingReportEvent::Ptr p_timing_event;
164  std::uint64_t vector_size;
165  std::uint64_t default_batch_size = m_benchmark_configuration.default_sample_size > 0 ?
166  m_benchmark_configuration.default_sample_size :
167  BenchmarkDescription::DefaultBatchSize;
168  std::uint64_t batch_sizes[BenchmarkDescription::OpParameterCount];
169  std::stringstream ss;
170 
171  hebench::APIBridge::BenchmarkDescriptor bench_desc = m_descriptor;
172  if (bench_desc.cat_params.offline.data_count[DataGenerator::Index_W] == 0)
173  bench_desc.cat_params.offline.data_count[DataGenerator::Index_W] = 1;
174  if (bench_desc.cat_params.offline.data_count[DataGenerator::Index_b] == 0)
175  bench_desc.cat_params.offline.data_count[DataGenerator::Index_b] = 1;
176 
177  vector_size = BenchmarkDescription::fetchVectorSize(m_params);
178  BenchmarkDescription::computeSampleSizes(batch_sizes,
179  BenchmarkDescription::OpParameterCount,
180  default_batch_size,
181  bench_desc);
182 
183  assert(batch_sizes[DataGenerator::Index_W] == 1 && batch_sizes[DataGenerator::Index_b] == 1);
184 
185  std::cout << IOS_MSG_INFO << hebench::Logging::GlobalLogger::log("Generating workload...") << std::endl;
186 
187  std::cout << IOS_MSG_INFO << hebench::Logging::GlobalLogger::log("Loading workload data...") << std::endl;
188 
189  DataGenerator::PolynomialDegree pd;
190  switch (m_descriptor.workload)
191  {
192  case hebench::APIBridge::Workload::LogisticRegression_PolyD3:
193  pd = DataGenerator::PolynomialDegree::PD3;
194  break;
195  case hebench::APIBridge::Workload::LogisticRegression_PolyD5:
196  pd = DataGenerator::PolynomialDegree::PD5;
197  break;
198  case hebench::APIBridge::Workload::LogisticRegression_PolyD7:
199  pd = DataGenerator::PolynomialDegree::PD7;
200  break;
201  default:
202  pd = DataGenerator::PolynomialDegree::None;
203  break;
204  } // end switch
205 
206  timer.start();
207  m_data = DataGenerator::create(pd,
208  vector_size,
209  batch_sizes[DataGenerator::Index_X],
210  m_descriptor.data_type);
211  p_timing_event = timer.stop<std::milli>();
212 
213  ss = std::stringstream();
214  ss << "Total data loaded: " << m_data->getTotalDataLoaded() << " bytes";
215  std::cout << IOS_MSG_DONE << hebench::Logging::GlobalLogger::log(ss.str()) << std::endl;
216  ss = std::stringstream();
217  ss << "Elapsed wall time: " << p_timing_event->elapsedWallTime<std::milli>() << " ms";
218  std::cout << IOS_MSG_INFO << hebench::Logging::GlobalLogger::log(ss.str()) << std::endl;
219  ss = std::stringstream();
220  ss << "Elapsed CPU time: " << p_timing_event->elapsedCPUTime<std::milli>() << " ms";
221  std::cout << IOS_MSG_INFO << hebench::Logging::GlobalLogger::log(ss.str()) << std::endl;
222 }
